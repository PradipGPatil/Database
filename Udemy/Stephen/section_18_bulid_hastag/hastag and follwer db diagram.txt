-- hashtag occure in post content,comment, and profile of the phototable users{
id serial [pk,increment]
created_at timestamp
updated_at timestamp
username varchar(40)
bio varchar(200)
avatar varchar(200)
phone varchar(25)
email varchar (50)
password varchar (200)
status varchar(20)

}

table posts{
  id serial [pk,increment]
  created_at timestamp
  updated_at timestamp
  url varchar(200)
  user_id integer [ref:> users.id]
  caption varchar(240)
  lat real
  lang real
}

table comments{
id serial [pk,increment]
created_at timestamp
updated_at timestamp
contents varchar(240)
post_id integer [ref: >posts.id]
user_id integer [ref:>users.id]
}

table likes{
  id serial [pk , increment]
  created_at timestamp

  post_id integer [ref:>posts.id]
  comment_id integer [ref:> comments.id]
   user_id integer [ref:> users.id]
}

  created_at timestamp

  post_id integer [ref:>users.id]
  user_id interger [ref:>users.id]

table captions_tag{
  id serial [pk , increment]

table hastags{
  id serial [pk, increment]

}

Table hastags_posts{
  id serial [pk,increment]
